    </td>
  </tr>

    <td> Actors </td>

      </ul>
    </td>
  </tr>

    <td>

          <li>The risk score calculation has run at least once</li>
      </ul>
    </td>
  </tr>
  <tr>
    <td> Basic Flow </td>
    <td>
      <ol>
          <li>User is in the View Map Screen</li>
          <li>User opens the Risk Heat Map View</li>

      </ol>
    </td>
  </tr>
  <tr>
    <td> Alternate/Exception Flows </td>
    <td>
      None
    </td>
  <tr>
    <td> Post Conditions </td>
    <td>

    <td>
  </tr>
